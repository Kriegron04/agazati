def sikeres(pontszam):
    if pontszam >= 48:
        return True
    else:
        return False

nev = input("Add meg a vizsgázó nevét: ")
while nev != "":
    pontszam = int(input("Add meg a pontszámát: "))
    if sikeres(pontszam):
