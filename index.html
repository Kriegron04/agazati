def sikeres(pontszam):
    if pontszam >= 48:
        return True
    else:
        return False

def legnagyobb_vasarlo(nevek, osszegek):
    max_osszeg = max(osszegek)
    index = osszegek.index(max_osszeg)
    return nevek[index]
